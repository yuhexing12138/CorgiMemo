# 待办页面"已完成"折叠区域 实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 移除待办页面顶部的"全部/待办/已完成"三筛选按钮，改为底部可折叠的"已完成 N"切换按钮，支持展开/折叠已完成待办列表，已完成项支持完整交互（拖拽排序、左滑、编辑），展开状态持久化。

**Architecture:** 在ViewModel层移除FilterStatus枚举，分离pendingTodos/completedTodos两个列表，保留filteredTodos（当前可见项：pending + 展开时的completed），新增showCompleted状态并持久化；在UI层引入DisplayItem密封类区分Todo项和CompletedDivider分隔项；扩展ReorderableLazyColumn支持isDraggable参数让分隔项不可拖拽不可跨越；CompletedDivider作为列表项在pending和completed之间渲染。

**Tech Stack:** Jetpack Compose, Kotlin Coroutines/Flow, Hilt, EncryptedSharedPreferences

```kotlin
    private val sortedFilteredTodos: StateFlow<List<TodoItem>> =
        kotlinx.coroutines.flow.combine(
            _todos, _searchQuery, _selectedCategoryId, _sortType
        ) { todos, query, categoryId, sortType ->
            var result = todos

            // 应用分类过滤
            if (categoryId != null && categoryId > 0) {
                result = result.filter { it.categoryId == categoryId }
            } else if (categoryId != null && categoryId == 0L) {
                val validCategoryIds = categories.value.map { it.id }.toSet()
                result = result.filter { it.categoryId !in validCategoryIds }
            }

            // 应用搜索关键词过滤（支持纯文本 + 格式化内容搜索）
            if (query.isNotBlank()) {
                result = result.filter { todo ->
                    todo.title.contains(query, ignoreCase = true) ||
                    (todo.content?.contains(query, ignoreCase = true) ?: false) ||
                    (todo.contentFormat?.let { format ->
                        com.corgimemo.app.util.MarkdownParser.stripMarkdown(format)
                            .contains(query, ignoreCase = true)
                    } ?: false)
                }
            }

            /** 应用排序 */
            when (sortType) {
                "updated_desc" -> result.sortedWith(
                    compareByDescending<TodoItem> { it.isPinned }
                        .thenBy { it.sortOrder }
                        .thenByDescending { it.updatedAt }
                )
                "updated_asc" -> result.sortedWith(
                    compareByDescending<TodoItem> { it.isPinned }
                        .thenBy { it.sortOrder }
                        .thenBy { it.updatedAt }
                )
                "created_desc" -> result.sortedWith(

                )
                "created_asc" -> result.sortedWith(
                    compareByDescending<TodoItem> { it.isPinned }
                        .thenBy { it.sortOrder }
                        .thenBy { it.createdAt }
                )
                else -> result.sortedWith(
                    compareByDescending<TodoItem> { it.isPinned }
                        .thenBy { it.sortOrder }
                        .thenByDescending { it.updatedAt }
                )
            }
        }.stateIn(viewModelScope, SharingStarted.Lazily, emptyList())

    /** 未完成待办列表（status=0） */
    val pendingTodos: StateFlow<List<TodoItem>> =
        sortedFilteredTodos.map { list -> list.filter { it.status == 0 } }

    /** 已完成待办列表（status=1，30天内） */
    private val visibleCompletedTodos: StateFlow<List<TodoItem>> =
        sortedFilteredTodos.map { list ->
```

- [ ] **Step 7: 重写reorderTodos为区域内排序方法**

删除原有的`reorderTodos`方法（第733-760行），替换为以下两个方法：

```kotlin
    /**
     * 未完成区域内拖拽排序
     * @param fromIndex 拖拽起始位置在pending列表中的索引
     * @param toIndex 拖拽目标位置在pending列表中的索引
     * @param crossedPinnedZone 是否跨越了置顶边界
     */
    fun reorderPendingTodos(fromIndex: Int, toIndex: Int, crossedPinnedZone: Boolean) {
        viewModelScope.launch {
            val pendingList = pendingTodos.value.toMutableList()
            if (fromIndex !in pendingList.indices || toIndex !in 0..pendingList.size) return@launch

            val draggedItem = pendingList.removeAt(fromIndex)
            val finalDraggedItem = if (crossedPinnedZone) {
                draggedItem.copy(isPinned = !draggedItem.isPinned)
            } else {
                draggedItem
            }
            val insertIndex = toIndex.coerceAtMost(pendingList.size)
            pendingList.add(insertIndex, finalDraggedItem)

            val completedList = visibleCompletedTodos.value
            recalculateGlobalSortOrder(pendingList, completedList, finalDraggedItem, crossedPinnedZone)
        }
    }

    /**
     * 已完成区域内拖拽排序
     * @param fromIndex 拖拽起始位置在completed列表中的索引
     * @param toIndex 拖拽目标位置在completed列表中的索引
     * @param crossedPinnedZone 是否跨越了置顶边界
     */
    fun reorderCompletedTodos(fromIndex: Int, toIndex: Int, crossedPinnedZone: Boolean) {
        viewModelScope.launch {
            val completedList = visibleCompletedTodos.value.toMutableList()
            if (fromIndex !in completedList.indices || toIndex !in 0..completedList.size) return@launch

            val draggedItem = completedList.removeAt(fromIndex)
            val finalDraggedItem = if (crossedPinnedZone) {
                draggedItem.copy(isPinned = !draggedItem.isPinned)
            } else {
                draggedItem
            }
            val insertIndex = toIndex.coerceAtMost(completedList.size)
            completedList.add(insertIndex, finalDraggedItem)

            val pendingList = pendingTodos.value
            recalculateGlobalSortOrder(pendingList, completedList, finalDraggedItem, crossedPinnedZone)
        }
    }

    /**
     * 重新计算全局sortOrder
     * 顺序：pending(pinned→normal) → completed(pinned→normal)
     */
    private suspend fun recalculateGlobalSortOrder(
        pendingList: List<TodoItem>,
        completedList: List<TodoItem>,
        draggedItem: TodoItem,
        crossedPinnedZone: Boolean
    ) {
        var globalIndex = 0

        // pending置顶项
        pendingList.filter { it.isPinned }.forEach { item ->
            if (item.sortOrder != globalIndex || (item.id == draggedItem.id && crossedPinnedZone)) {
                todoRepository.updateSortOrder(item.id, globalIndex)
                if (item.id == draggedItem.id && crossedPinnedZone) {
                    todoRepository.updatePinnedStatus(item.id, item.isPinned)
                }
            }
            globalIndex++
        }
        // pending普通项
        pendingList.filter { !it.isPinned }.forEach { item ->
            if (item.sortOrder != globalIndex || (item.id == draggedItem.id && crossedPinnedZone)) {
                todoRepository.updateSortOrder(item.id, globalIndex)
                if (item.id == draggedItem.id && crossedPinnedZone) {
                    todoRepository.updatePinnedStatus(item.id, item.isPinned)
                }
            }
            globalIndex++
        }
        // completed置顶项
        completedList.filter { it.isPinned }.forEach { item ->
            if (item.sortOrder != globalIndex || (item.id == draggedItem.id && crossedPinnedZone)) {
                todoRepository.updateSortOrder(item.id, globalIndex)
                if (item.id == draggedItem.id && crossedPinnedZone) {
                    todoRepository.updatePinnedStatus(item.id, item.isPinned)
                }
            }
            globalIndex++
        }
        // completed普通项
        completedList.filter { !it.isPinned }.forEach { item ->
            if (item.sortOrder != globalIndex || (item.id == draggedItem.id && crossedPinnedZone)) {
                todoRepository.updateSortOrder(item.id, globalIndex)
                if (item.id == draggedItem.id && crossedPinnedZone) {
                    todoRepository.updatePinnedStatus(item.id, item.isPinned)
                }
            }
            globalIndex++
        }
    }
```

```kotlin
    items: List<T>,
```

```kotlin
    draggedTop: Float,
    draggedCenter: Float,
    draggedBottom: Float,
    isPinned: (T) -> Boolean,
    isDraggable: (T) -> Boolean
): Int {
    val draggedItemIsPinned = isPinned(items[draggedIndex])

    for (childInfo in layoutInfo.visibleItemsInfo) {
        val index = childInfo.index
        if (index == draggedIndex) continue

        if (draggedItemIsPinned && !isPinned(item)) continue
        if (!draggedItemIsPinned && isPinned(item)) continue
```
