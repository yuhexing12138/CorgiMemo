```kotlin
    /**
     * 拼图数量超过上限异常
     *
     * @param count 实际拼图数量
     */
    class TooManyBitmapsException(val count: Int) :
        IllegalArgumentException("拼图数量 $count 超过上限 10，请选择一条条分享")

    /**
     * 拼图策略
     */
    private enum class MergeStrategy {
        GRID_2_COL,  // 2-6 张：2 列网格
        VERTICAL     // 7-10 张：垂直拼接
    }

    /**
     * 合并多张待办分享图片为一张
     *
     * 智能拼接策略：
     * - 2-6 张：2 列网格布局
     * - 7-10 张：垂直拼接
     * - >10 张：抛出 [TooManyBitmapsException]（调用方需禁用合并按钮）
     * - 0 张：抛出 [IllegalArgumentException]
     *
     * 所有图片宽度按比例缩放至 720dp，高度按比例计算。
     * 拼接背景为纯白，子图间隔 24 像素。
     *
     * @param bitmaps 待合并的 Bitmap 列表（已按顺序排列）
     * @param outputWidthPx 输出图片宽度（像素），默认 720
     * @param spacingPx 子图间隔像素，默认 24
     * @return 合并后的 Bitmap
     * @throws TooManyBitmapsException 当 count > 10
     * @throws IllegalArgumentException 当 count == 0
     */
    suspend fun mergeBitmaps(
        bitmaps: List<Bitmap>,
        outputWidthPx: Int = 720,
        spacingPx: Int = 24
    ): Bitmap = withContext(Dispatchers.Default) {
        require(bitmaps.isNotEmpty()) { "拼图列表不能为空" }
        if (bitmaps.size > 10) {
            throw TooManyBitmapsException(bitmaps.size)
        }

        val strategy = if (bitmaps.size <= 6) MergeStrategy.GRID_2_COL else MergeStrategy.VERTICAL

        // 缩放所有子图至 outputWidthPx（或一半宽用于网格列）
        val columnWidth: Int
        val rowScaledBitmaps: List<Bitmap>

        when (strategy) {
            MergeStrategy.GRID_2_COL -> {
                // 网格：每列宽 = (outputWidthPx - spacing) / 2
                columnWidth = (outputWidthPx - spacingPx) / 2
                rowScaledBitmaps = bitmaps.map { scaleBitmapToWidth(it, columnWidth) }
            }
            MergeStrategy.VERTICAL -> {
                // 垂直：单列宽 = outputWidthPx
                columnWidth = outputWidthPx
                rowScaledBitmaps = bitmaps.map { scaleBitmapToWidth(it, columnWidth) }
            }
        }

        val columnsCount = if (strategy == MergeStrategy.GRID_2_COL) 2 else 1
        val rowsCount = (rowScaledBitmaps.size + columnsCount - 1) / columnsCount

        // 计算每行高度（取该行最高子图）
        val rowHeights = IntArray(rowsCount) { rowIdx ->
            val start = rowIdx * columnsCount
            val end = minOf(start + columnsCount, rowScaledBitmaps.size)
            (start until end).maxOf { rowScaledBitmaps[it].height }
        }

        val totalHeight = rowHeights.sum() + (rowsCount - 1) * spacingPx
        val totalWidth = outputWidthPx

        val output = Bitmap.createBitmap(totalWidth, totalHeight, Bitmap.Config.ARGB_8888)
        val canvas = Canvas(output)
        canvas.drawColor(Color.WHITE)

        // 按行绘制
        var currentY = 0
        for (rowIdx in 0 until rowsCount) {
            val start = rowIdx * columnsCount
            val end = minOf(start + columnsCount, rowScaledBitmaps.size)
            val rowHeight = rowHeights[rowIdx]

            for (colIdx in 0 until (end - start)) {
                val bmp = rowScaledBitmaps[start + colIdx]
                val x = colIdx * (columnWidth + spacingPx)
                // 垂直居中
                val y = currentY + (rowHeight - bmp.height) / 2
                canvas.drawBitmap(bmp, x.toFloat(), y.toFloat(), null)
            }
            currentY += rowHeight + spacingPx
        }

        output
    }

    /**
     * 按宽度等比缩放 Bitmap
     */
    private fun scaleBitmapToWidth(source: Bitmap, targetWidth: Int): Bitmap {
        if (source.width == targetWidth) return source
        val ratio = targetWidth.toFloat() / source.width
        val targetHeight = (source.height * ratio).toInt()
        return Bitmap.createScaledBitmap(source, targetWidth, targetHeight, true)
    }
```

```kotlin
// app/src/main/java/com/corgimemo/app/backup/exporter/ShareCoordinator.kt
package com.corgimemo.app.backup.exporter

import android.content.Context
import android.widget.Toast
import com.corgimemo.app.data.model.Category
import com.corgimemo.app.data.model.TodoItem
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.delay
import kotlinx.coroutines.withContext

/**
 * 分享协调器（统一入口）
 *
 * 负责协调"待办编辑页"和"多选模式"两处分享入口的逻辑：
 * - 判断单/多个待办
 * - 单个：直接调 [shareTodoAsImage]（单张分享）
 * - 多个：通过 [onShowDialog] 回调显示选择弹窗（合并/一条条）
 * - 未保存分组：通过 [onShowSnackBar] 提示用户先保存
 *
 * 本身不持有 UI 状态，UI 层传入回调即可。
 */
object ShareCoordinator {

    /**
     * 从待办编辑页调用
     *
     * @param context 上下文（用于启动 Intent）
     * @param mainTodo 主 todo（编辑页主对象）
     * @param savedSubTodos 已保存的子 todo 列表（按 groupId 拆分）
     * @param categories 分类列表
     * @param hasUnsavedGroups 是否有未保存的分组（true 时不进入分享流程）
     * @param onShowSnackBar 显示 SnackBar（用于未保存提示）
     * @param onShowDialog 显示分享方式选择弹窗（参数为待分享数量）
     */
    suspend fun shareTodosFromEdit(
        context: Context,
        mainTodo: TodoItem,
        savedSubTodos: List<TodoItem>,
        categories: List<Category>,
        hasUnsavedGroups: Boolean,
        onShowSnackBar: (String) -> Unit,
        onShowDialog: ((count: Int) -> Unit)? = null
    ) {
        if (hasUnsavedGroups) {
            onShowSnackBar("请先保存所有分组")
            return
        }

        val allTodos = listOf(mainTodo) + savedSubTodos
        if (allTodos.isEmpty()) return

        if (allTodos.size == 1) {
            shareTodoAsImage(context, allTodos[0], categories)
            return
        }

        // 多个 → 弹选择弹窗
        onShowDialog?.invoke(allTodos.size)
    }

    /**
     * 从多选模式调用
     *
     * @param context 上下文
     * @param todos 选中的 todo 列表
     * @param categories 分类列表
     * @param onShowDialog 显示分享方式选择弹窗
     */
    suspend fun shareTodos(
        context: Context,
        todos: List<TodoItem>,
        categories: List<Category>,
        onShowDialog: ((count: Int) -> Unit)? = null
    ) {
        if (todos.isEmpty()) return

        if (todos.size == 1) {
            shareTodoAsImage(context, todos[0], categories)
            return
        }

        onShowDialog?.invoke(todos.size)
    }

    /**
     * 用户在弹窗中选择了"合并分享"
     *
     * @param context 上下文
     * @param todos 待分享的 todo 列表
     * @param categories 分类列表
     * @param onShowSnackBar 用于错误提示
     */
    suspend fun shareMerged(
        context: Context,
        todos: List<TodoItem>,
        categories: List<Category>,
        onShowSnackBar: (String) -> Unit
    ) {
        try {
            withContext(Dispatchers.IO) {
                val bitmaps = todos.map { todo ->
                    val category = categories.find { it.id == todo.categoryId }
                    ImageExporter.createTodoShareCard(context, todo, category)
                }
                val merged = ImageExporter.mergeBitmaps(bitmaps)
                val file = ImageExporter.saveBitmapToCache(context, merged)
                ShareIntentHelper.shareImage(
                    context = context,
                    imageFile = file,
                    text = "我在 CorgiMemo 创建了 ${todos.size} 条待办"
                )
            }
        } catch (e: ImageExporter.TooManyBitmapsException) {
            onShowSnackBar("图片过多，请选择一条条分享")
        } catch (e: Exception) {
            Toast.makeText(context, "图片生成失败: ${e.message}", Toast.LENGTH_SHORT).show()
        }
    }

    /**
     * 用户在弹窗中选择了"一条条分享"
     *
     * @param context 上下文
     * @param todos 待分享的 todo 列表
     * @param categories 分类列表
     */
    suspend fun shareOneByOne(
        context: Context,
        todos: List<TodoItem>,
        categories: List<Category>
    ) {
        for (todo in todos) {
            try {
                shareTodoAsImage(context, todo, categories)
                delay(800)  // 避免系统分享面板叠加
            } catch (e: Exception) {
                Toast.makeText(context, "分享失败: ${e.message}", Toast.LENGTH_SHORT).show()
            }
        }
    }
}
```

```kotlin
package com.corgimemo.app.ui.components

import androidx.compose.foundation.layout.Column
import androidx.compose.foundation.layout.Spacer
import androidx.compose.foundation.layout.height
import androidx.compose.material3.AlertDialog
import androidx.compose.material3.Button
import androidx.compose.material3.ButtonDefaults

import androidx.compose.ui.unit.dp
import com.corgimemo.app.R

/**
 * 分享方式选择弹窗
 *
 * 当用户需要分享多个待办时显示，让用户在"合并为一张"和"一条条"之间选择。
 *
 * @param count 待分享的待办数量
 * @param enableMerge 是否启用"合并分享"按钮（>10 时为 false）
 * @param onDismiss 关闭弹窗
 * @param onMerge 选择"合并分享"
 * @param onOneByOne 选择"一条条分享"
 */
@Composable
fun ShareModeDialog(
    count: Int,
    enableMerge: Boolean,
    onDismiss: () -> Unit,
    onMerge: () -> Unit,
    onOneByOne: () -> Unit
) {
    AlertDialog(
        onDismissRequest = onDismiss,
        title = { Text(text = stringResource(id = R.string.share_mode_title)) },
        text = {
            Column {
                Text(
                    text = if (enableMerge) {
                        stringResource(id = R.string.share_mode_subtitle_normal, count)
                    } else {
                        stringResource(id = R.string.share_mode_subtitle_too_many, count)
                    },
                    style = MaterialTheme.typography.bodyMedium
                )
            }
        },
        confirmButton = {
            Button(
```

```kotlin
    /**
     * 获取当前 todo 中所有"已保存"的子 todo 列表
     *
     * 用于分享功能：从 [groupSaveStates] 中筛选 isSaved==true 的分组，
     * 通过 [TodoRepository] 查询对应的子 todo。
     *
     * @return 已保存的子 todo 列表（按 groupId 升序）
     */
    suspend fun getSavedSubTodos(): List<TodoItem> {
        val savedGroupIds = _groupSaveStates.value
            .filterValues { it.isSaved && it.savedTodoId != null }
            .keys
            .sorted()

        val result = mutableListOf<TodoItem>()
        for (groupId in savedGroupIds) {
            val savedTodoId = _groupSaveStates.value[groupId]?.savedTodoId ?: continue
            val todo = todoRepository.getTodoById(savedTodoId)
            if (todo != null) {
                result.add(todo)
            }
        }
        return result
    }
```
