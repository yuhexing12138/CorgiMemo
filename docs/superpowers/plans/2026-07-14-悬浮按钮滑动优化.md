- [ ] **Step 2: 实现新的 snapToEdge(基于像素 + spring 动画)**

新的 `snapToEdge` 需访问 `containerWidthPx`/`minX`/`maxX`/`offsetX`/`offsetY`,因此**必须定义在 BoxWithConstraints 内部**。

在 `LaunchedEffect(initialPosition, ...)` 之后、`Surface(...)` 之前插入:

```kotlin
        /**
         * 贴边吸附:根据当前 offsetX 判断吸附到左/右边缘,用 spring 平滑动画过渡
         * Y 坐标保持不变。完成后持久化百分比坐标到 DataStore。
         */
        fun snapToEdge() {
            val halfWidth = containerWidthPx / 2
            val targetX = if (offsetX < halfWidth) minX else maxX

            coroutineScope.launch {
                playBounceAnimation()
                // spring 平滑过渡
                val animX = Animatable(offsetX.toFloat())
                animX.animateTo(
                    targetValue = targetX,
                    animationSpec = spring(
                        dampingRatio = Spring.DampingRatioMediumBouncy,
                        stiffness = Spring.StiffnessMediumLow
                    )
                )
                offsetX = animX.value.toInt()

                // 持久化(百分比坐标,与 DataStore 旧格式兼容)
                val percentX = offsetX / containerWidthPx
                val percentY = 1f - (offsetY / containerHeightPx)
                onPositionChanged(percentX, percentY)
            }
        }
```

定位到 Task 1 Step 6 后的 `.pointerInput(Unit) {` 块,将其中从 `awaitPointerEventScope {` 开始到对应 `}` 结束的整个手势处理逻辑(L260-353 区域,已被 Task 1 改动后行号会变)替换为以下新版本:

```kotlin
                    // 长按阈值:200ms(不再为快速滑动让步,降低误触)
                    val LONG_PRESS_THRESHOLD = 200L
                    // 拖动启动位移阈值:8px(超过即视为拖动,响应更快)
                    val DRAG_START_THRESHOLD = 8f
                    // 点击位移阈值:8px(小于此值视为点击)
                    val CLICK_THRESHOLD = 8f

                    awaitPointerEventScope {
                        while (true) {
                            val down = awaitPointerEvent()
                            val downChange = down.changes.firstOrNull() ?: continue

                            val downX = downChange.position.x
                            val downY = downChange.position.y

                            // 命中区域判断:按下点必须在按钮矩形内,否则不消费事件
                            val insideButton = downX in offsetX..(offsetX + buttonSizePx) &&
                                               downY in offsetY..(offsetY + buttonSizePx)
                            if (!insideButton) continue

                            // 按下时记录手指相对按钮左上角的偏移(用于绝对位置追踪)
                            val touchOffsetX = downX - offsetX
                            val touchOffsetY = downY - offsetY

                            val dragStartTime = System.currentTimeMillis()
                            var totalDragX = 0f
                            var totalDragY = 0f
                            // 局部拖动标志:控制手势循环内逻辑流
                            // 注意:与外层可观察状态 isDragging 同名会遮蔽,故用 isDragActive
                            var isDragActive = false

                            while (true) {
                                val event = awaitPointerEvent()
                                val change = event.changes.firstOrNull() ?: break
                                if (change.isConsumed) break

                                val currentX = change.position.x
                                val currentY = change.position.y

                                if (!isDragActive) {
                                    // 累计相对按下点的总位移(用于点击 vs 拖动判定)
                                    totalDragX = currentX - downX
                                    totalDragY = currentY - downY

                                    val elapsed = System.currentTimeMillis() - dragStartTime
                                    val moved = abs(totalDragX) > DRAG_START_THRESHOLD ||
                                                abs(totalDragY) > DRAG_START_THRESHOLD
                                    // 长按超时 或 位移超阈值,任一满足即进入拖动
                                    if (elapsed > LONG_PRESS_THRESHOLD || moved) {
                                        isDragActive = true
                                        isDragging = true  // 同步外层可观察状态,触发 scale/shadow 变化
                                    }
                                }

                                if (isDragActive) {
                                    // 绝对位置追踪:手指位置 - 按下偏移 = 按钮目标位置
                                    // 无截断、无累积误差,丢帧后下一帧立即追上手指
                                    val targetX = (currentX - touchOffsetX)
                                        .coerceIn(minX, maxX)
                                    val targetY = (currentY - touchOffsetY)
                                        .coerceIn(minY, maxY)
                                    offsetX = targetX.toInt()
                                    offsetY = targetY.toInt()
                                    change.consume()
                                }

                                if (!change.pressed) {
                                    // 手指抬起
                                    if (isDragActive) {
                                        // 拖动结束:贴边吸附
                                        isDragActive = false
                                        isDragging = false
                                        snapToEdge()
                                    } else if (abs(totalDragX) < CLICK_THRESHOLD &&
                                               abs(totalDragY) < CLICK_THRESHOLD) {
                                        // 点击:先播放缩放动画再回调
                                        coroutineScope.launch {
                                            playClickAnimation()
                                            onClick()
                                        }
                                    }
                                    break
                                }
                            }
                        }
                    }
```
