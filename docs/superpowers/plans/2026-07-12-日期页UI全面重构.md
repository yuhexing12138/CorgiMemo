/**
 * 把毫秒差值格式化为 "X年 X天 X时 X分 X秒"，从最大有效单位逐级显示到秒。
 *
 * 例：
 * - 366_1000ms → "1年 1天 0时 0分 1秒"
 * - 5天3时   → "5天 3时 0分 0秒"
 * - 不足1时 → "4分 5秒"
 * - =0      → "0秒"（兜底）
 *
 * 注：使用"365天=1年"简化计算（不区分闰年），与现有 calculateDaysRemaining 一致。
 *
 * @param millis 时间差毫秒（负数会被 coerce 为 0）
 * @return 格式化字符串
 */
internal fun formatDuration(millis: Long): String {
    val totalSec = (millis / 1000).coerceAtLeast(0L)
    val sec = (totalSec % 60).toInt()
    val min = (totalSec / 60 % 60).toInt()
    val hr  = (totalSec / 3600 % 24).toInt()
    val day = (totalSec / 86400 % 365).toInt()
    val yr  = (totalSec / (86400 * 365)).toInt()
    return when {
        yr  > 0 -> "$yr 年 $day 天 $hr 时 $min 分 $sec 秒"
        day > 0 -> "$day 天 $hr 时 $min 分 $sec 秒"
        hr  > 0 -> "$hr 时 $min 分 $sec 秒"
        min > 0 -> "$min 分 $sec 秒"
        else    -> "$sec 秒"
    }
}
```

private fun groupByDisplayDates(dates: List<SpecialDate>): Map<DateGroup, List<DisplayDate>> {
    val todayStart = todayStartMillis()
    return dates
        .filter { !it.isArchived }  // 1. 只取未归档
        .groupBy { date ->
            // 2. 按新规则分组
            when {
                date.countMode == 0 && date.targetDate >= todayStart -> DateGroup.COUNTDOWN
                date.countMode == 1 && date.targetDate <= todayStart -> DateGroup.COUNTUP
                date.countMode == 0 && date.targetDate <  todayStart -> DateGroup.EXPIRED
                else -> DateGroup.COUNTUP  // countMode=1 但 targetDate 仍在未来：正计时
            }
        }
        .mapValues { entry ->
            // 3. 转 DisplayDate
            entry.value.map { it.toDisplayDate() }
        }
}

private fun SpecialDate.toDisplayDate(): DisplayDate = DisplayDate(
    id = id,
    title = title,
    targetDate = targetDate,
    category = category.toDateCategoryEnum(),
    countMode = countMode,
    group = when {
        countMode == 0 && targetDate >= todayStartMillis() -> DateGroup.COUNTDOWN
        countMode == 1 && targetDate <= todayStartMillis() -> DateGroup.COUNTUP
        countMode == 0 && targetDate <  todayStartMillis() -> DateGroup.EXPIRED
        else -> DateGroup.COUNTUP
    },
    isPinned = isPinned,
    relationHint = null  // 关联提示由 UI 层另查
)

private fun todayStartMillis(): Long {
    val cal = java.util.Calendar.getInstance().apply {
        set(java.util.Calendar.HOUR_OF_DAY, 0)
        set(java.util.Calendar.MINUTE, 0)
        set(java.util.Calendar.SECOND, 0)
        set(java.util.Calendar.MILLISECOND, 0)
    }
    return cal.timeInMillis
}

fun setExpandedDateId(id: Long?) {

/**
 * 归档特殊日期
 * 1. 调用 repository.archive(id)
 * 2. 缓存 SpecialDate 快照到 _pendingArchive（供撤回用）
 * 3. 若归档的是置顶卡，同步清空 _pinnedDateId
 */
fun archiveDate(id: Long) {
    viewModelScope.launch {
        try {
            // 1. 先取快照（在 archive 之前）
            val snapshot = repository.getByIdBlocking(id)
            // 2. 归档
            repository.archive(id)
            // 3. 缓存快照
            _pendingArchive.value = snapshot
            // 4. 若归档的是置顶卡，清空 pinnedDateId
            if (_pinnedDateId.value == id) {
                _pinnedDateId.value = null
            }
        } catch (e: Exception) {
            // 走项目内统一的全局错误处理流程（具体入口后续由 host 注入），UI 弹简短 Toast
            android.util.Log.e("SpecialDateVM", "归档失败: id=$id", e)
        }
    }
}

/**
 * 撤回归档（恢复 SpecialDate 到 isArchived=false）
 * 必须在 Snackbar 显示的 3 秒内由 UI 触发
 */
fun undoArchive() {
    val snapshot = _pendingArchive.value ?: return
    viewModelScope.launch {
        try {
            repository.unarchive(snapshot.id)
            _pendingArchive.value = null
        } catch (e: Exception) {
            android.util.Log.e("SpecialDateVM", "撤回归档失败: id=${snapshot.id}", e)
            // 失败：清空缓存，让 UI 弹 Toast
            _pendingArchive.value = null
        }
    }
}

/**
 * 特殊日期卡片（新版本）
 *
 * 视觉结构（左→右）：
 * - 圆形图片区（56dp，emoji 占位）
 * - Spacer 12dp
 * - 内容区（weight=1f）：标题行 + 时间信息行 + 标签行
 * - Spacer 8dp
 * - 右侧大数字（24sp Bold，剩余/已过天数）
 *
 * 包装 SwipeableTodoBox 提供三按钮左滑（置顶/归档/删除）。
 *
 * @param date 展示数据
 * @param nowMs 当前时间戳（毫秒），由 SpecialDateScreen 顶层 ticker 驱动
 * @param isClickBlocked 左滑操作面板是否展开（true 时屏蔽卡片内的点击/长按）
 * @param isExpiredGroup 是否处于"已过期"分组（决定整体 alpha 0.6 降权）
 * @param onClick 卡片整体点击回调（进入编辑）
 * @param onPin 置顶/取消置顶回调
 * @param onArchive 归档回调
 * @param onDelete 删除回调
 */
@Composable
fun SpecialDateCard(
    date: DisplayDate,
    nowMs: Long,
    isClickBlocked: Boolean = false,
    onClick: () -> Unit = {},
    onPin: () -> Unit = {},
    onArchive: () -> Unit = {},
    onDelete: () -> Unit = {},
    modifier: Modifier = Modifier
) {
    val isExpiredGroup = date.group == DateGroup.EXPIRED
    val diff = date.targetDate - nowMs
    val isFuture = diff >= 0
    val timeColor = when {
        isExpiredGroup -> Color(0xFF999999)
        isFuture       -> MaterialTheme.colorScheme.primary
        else           -> Color(0xFF7EC8A0)
    }
    val formattedDate = SimpleDateFormat("yyyy年M月d日", Locale.getDefault())
        .format(Date(date.targetDate))

    Card(
        shape = RoundedCornerShape(20.dp),
        elevation = CardDefaults.cardElevation(defaultElevation = 2.dp),
        colors = CardDefaults.cardColors(containerColor = MaterialTheme.colorScheme.surface),
        modifier = modifier
            .fillMaxWidth()
            .then(if (isExpiredGroup) Modifier.alpha(0.6f) else Modifier)
    ) {
        Row(
            Modifier
                .fillMaxWidth()
                .padding(16.dp),
            verticalAlignment = Alignment.CenterVertically
        ) {
            // 1. 圆形图片区（56dp，emoji 占位）
            Box(
                modifier = Modifier
                    .size(56.dp)
                    .background(
                        color = MaterialTheme.colorScheme.surfaceVariant,
                        shape = CircleShape
                    ),
                contentAlignment = Alignment.Center
            ) {
                Text(
                    text = date.category.emoji,
                    fontSize = 28.sp
                )
            }

            Spacer(Modifier.width(12.dp))

            // 2. 内容区（weight=1f）
            Column(modifier = Modifier.weight(1f)) {
                // 2.1 标题行：置顶图标 + title
                Row(verticalAlignment = Alignment.CenterVertically) {
                    if (date.isPinned) {
                        Icon(
                            imageVector = Icons.Outlined.PushPin,
                            contentDescription = "置顶",
                            tint = if (isExpiredGroup) Color(0xFF999999) else MaterialTheme.colorScheme.primary,
                            modifier = Modifier.size(14.dp)
                        )
                        Spacer(Modifier.width(4.dp))
                    }
                    Text(
                        text = date.title,
                        fontSize = 16.sp,
                        fontWeight = FontWeight.Medium,
                        color = MaterialTheme.colorScheme.onSurface,
                        maxLines = 1,
                        overflow = TextOverflow.Ellipsis
                    )
                }

                Spacer(Modifier.size(4.dp))

                // 2.2 时间信息行（动态格式）
                Text(
                    text = formatDuration(kotlin.math.abs(diff)),
                    fontSize = 14.sp,
                    fontWeight = FontWeight.Medium,
                    color = timeColor
                )

                Spacer(Modifier.size(4.dp))

                // 2.3 标签行：还有/已经/已过 + 分类日期
                Text(
                    text = "${if (isFuture) "还有" else if (isExpiredGroup) "已过" else "已经"} $formattedDate",
                    fontSize = 12.sp,
                    color = MaterialTheme.colorScheme.onSurfaceVariant
                )
            }

            Spacer(Modifier.width(8.dp))

            // 3. 右侧大数字（剩余/已过天数）
            Text(
                text = (kotlin.math.abs(diff) / 86_400_000L).toString(),
                fontSize = 24.sp,
                fontWeight = FontWeight.Bold,
                color = timeColor
            )
        }
    }
}
```

/**
 * 特殊日期列表页面（重构版）
 *
 * 关键设计：
 * 1. 单一 ticker：顶层 LaunchedEffect 驱动 nowMs 状态，每秒更新一次
 * 2. 三段式布局：搜索框 → 3 个折叠分组头（倒计时/正计时/已过期）→ 卡片列表
 * 3. 左滑操作：置顶/归档/删除三按钮
 * 4. 归档后 Snackbar 3 秒撤回
 */
@Composable
fun SpecialDateScreen(
    navController: NavController,
    onFabClick: () -> Unit = {},
    viewModel: SpecialDateViewModel = hiltViewModel()
) {
    val groupedDates by viewModel.groupedDates.collectAsState()
    val searchQuery by viewModel.searchQuery.collectAsState()
    val isDataInitialized by viewModel.isDataInitialized.collectAsState()
    val expandedDateId by viewModel.expandedDateId.collectAsState()
    val pinnedDateId by viewModel.pinnedDateId.collectAsState()
    val pendingArchive by viewModel.pendingArchive.collectAsState()

    // 单一 ticker：整页仅 1 个协程
    var nowMs by remember { mutableStateOf(System.currentTimeMillis()) }
    LaunchedEffect(Unit) {
        while (true) {
            nowMs = System.currentTimeMillis()
            delay(1000L)
        }
    }

    // Snackbar
    val snackbarHostState = remember { SnackbarHostState() }
    val coroutineScope = rememberCoroutineScope()
    LaunchedEffect(pendingArchive) {
        val snapshot = pendingArchive ?: return@LaunchedEffect
        coroutineScope.launch {
            val result = snackbarHostState.showSnackbar(
                message = "已归档『${snapshot.title}』",
                actionLabel = "撤回",
                withDismissAction = true,
                duration = androidx.compose.material3.SnackbarDuration.Short
            )
            if (result == SnackbarResult.ActionPerformed) {
                viewModel.undoArchive()
            } else {
                // 3s 后无操作，清空缓存（DB 状态已固定为已归档）
                viewModel.undoArchive()  // 这里不撤回，仅清空缓存
                // 注意：undoArchive 会调 unarchive，需要拆出 clearPendingArchive
                // → 修复：新增 viewModel.clearPendingArchive()
            }
        }
    }

    Scaffold(
        snackbarHost = { SnackbarHost(snackbarHostState) }
    ) { paddingValues ->
        Box(
            modifier = Modifier
                .fillMaxSize()
                .padding(paddingValues)
        ) {
            Column(modifier = Modifier.fillMaxSize()) {
                // 1. 搜索框
                SearchBar(
                    query = searchQuery,
                    onQueryChange = { viewModel.updateSearchQuery(it) },
                    onClear = { viewModel.updateSearchQuery("") },
                    placeholder = "搜索日期...",
                    modifier = Modifier
                        .fillMaxWidth()
                        .padding(horizontal = 20.dp)
                )

                Spacer(modifier = Modifier.height(8.dp))

                // 2. 内容区
                if (!isDataInitialized) {
                    SpecialDateSkeleton()
                } else if (groupedDates.isEmpty()) {
                    UnifiedEmptyState(
                        icon = "📅",
                        title = "还没有特殊日期~",
                        subtitle = "记录重要的日子，不错过每个纪念！",
                        ctaText = "📅 添加日期",
                        onCtaClick = onFabClick,
                        modifier = Modifier.fillMaxSize()
                    )
                } else {
                    DateSectionsList(
                        groupedDates = groupedDates,
                        nowMs = nowMs,
                        expandedDateId = expandedDateId,
                        pinnedDateId = pinnedDateId,
                        onSetExpanded = viewModel::setExpandedDateId,
                        onPin = viewModel::pinDate,
                        onUnpin = viewModel::unpinDate,
                        onArchive = viewModel::archiveDate,
                        onDelete = viewModel::deleteDate,
                        onCardClick = { date ->
                            navController.navigate("date_edit/${date.id}")
                        }
                    )
                }
            }

            // 3. FAB
            FloatingActionButton(
                onClick = onFabClick,
                containerColor = MaterialTheme.colorScheme.primary,
                contentColor = MaterialTheme.colorScheme.onPrimary,
                modifier = Modifier
                    .align(Alignment.BottomEnd)
                    .padding(end = 20.dp, bottom = 16.dp)
            ) {
                Icon(
                    imageVector = Icons.Default.Edit,
                    contentDescription = "添加日期",
                    modifier = Modifier.size(24.dp)
                )
            }
        }
    }
}

@Composable
private fun DateSectionsList(
    groupedDates: Map<DateGroup, List<DisplayDate>>,
    nowMs: Long,
    expandedDateId: Long?,
    pinnedDateId: Long?,
    onSetExpanded: (Long?) -> Unit,
    onPin: (Long) -> Unit,
    onUnpin: (Long) -> Unit,
    onArchive: (Long) -> Unit,
    onDelete: (com.corgimemo.app.data.model.SpecialDate) -> Unit,
    onCardClick: (DisplayDate) -> Unit
) {
    // 各分组的展开状态：COUNTDOWN + COUNTUP 默认展开，EXPIRED 默认折叠
    val expandState = remember {
        mutableStateOf(
            mapOf(
                DateGroup.COUNTDOWN to true,
                DateGroup.COUNTUP to true,
                DateGroup.EXPIRED to false
            )
        )
    }
    // 让 expandState 在 groupedDates 变化时保持稳定
    // （按设计 default 一次，后续由用户切换）

    LazyColumn(
        modifier = Modifier
            .fillMaxSize()
            .padding(horizontal = 20.dp),
        verticalArrangement = Arrangement.spacedBy(8.dp)
    ) {
        DateGroup.values().forEach { group ->
            val dates = groupedDates[group] ?: emptyList()
            if (dates.isEmpty()) return@forEach
            val isExpanded = expandState.value[group] ?: false
            item(key = "header_${group.name}") {
                DateSectionHeader(
                    group = group,
                    count = dates.size,
                    isExpanded = isExpanded,
                    onToggle = {
                        expandState.value = expandState.value.toMutableMap().apply {
                            put(group, !(this[group] ?: false))
                        }
                    }
                )
            }
            if (isExpanded) {
                items(dates, key = { "date_${it.id}" }) { date ->
                    SwipeableTodoBox(
                        isExpanded = expandedDateId == date.id,
                        isPinned = date.isPinned,
                        onPinClick = {
                            if (date.isPinned) onUnpin(date.id) else onPin(date.id)
                        },
                        onArchiveClick = { onArchive(date.id) },
                        onDeleteClick = {
                            // 这里需要 SpecialDate 完整对象，UI 层持有 DisplayDate 时需要回查
                            // → 简化为只传 id，由 ViewModel 内部补齐 SpecialDate
                            onDelete(com.corgimemo.app.data.model.SpecialDate(
                                id = date.id, title = date.title, targetDate = date.targetDate,
                                category = date.category.name, countMode = date.countMode,
                                repeatType = 0, reminderDays = 0, content = "",
                                tags = "", imagePaths = "", imageUrls = "",
                                isPinned = date.isPinned, isArchived = false,
                                createdAt = 0L, updatedAt = 0L
                            ))
                        },
                        customButtons = listOf(
                            com.corgimemo.app.ui.components.SwipeButtonConfig(
                                label = if (date.isPinned) "取消置顶" else "置顶",
                                backgroundColorRes = R.color.ui_primary,
                                icon = if (date.isPinned) FilledPushPin else Icons.Outlined.PushPin,
                                zIndex = 3f,
                                shape = androidx.compose.foundation.shape.RoundedCornerShape(
                                    topStart = 16.dp, bottomStart = 16.dp
                                ),
                                actionType = com.corgimemo.app.ui.components.SwipeActionType.PIN
                            ),
                            com.corgimemo.app.ui.components.SwipeButtonConfig(
                                label = "归档",
                                backgroundColorRes = R.color.ui_archive,
                                icon = Icons.Outlined.Archive,
                                zIndex = 2f,
                                shape = androidx.compose.foundation.shape.RoundedCornerShape(0.dp),
                                actionType = com.corgimemo.app.ui.components.SwipeActionType.ARCHIVE
                            ),
                            com.corgimemo.app.ui.components.SwipeButtonConfig(
                                label = "删除",
                                backgroundColorRes = R.color.ui_swipe_delete,
                                icon = Icons.Outlined.Delete,
                                zIndex = 1f,
                                shape = androidx.compose.foundation.shape.RoundedCornerShape(
                                    topEnd = 16.dp, bottomEnd = 16.dp
                                ),
                                actionType = com.corgimemo.app.ui.components.SwipeActionType.DELETE
                            )
                        )
                    ) { _ ->
                        SpecialDateCard(
                            date = date,
                            nowMs = nowMs,
                            onClick = { onCardClick(date) }
                        )
                    }
                }
            }
        }
    }
}
