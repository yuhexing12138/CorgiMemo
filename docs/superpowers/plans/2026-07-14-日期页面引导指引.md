```kotlin
    /**
     * 创建首个日期
     * 在引导流程中创建并保存到 Room 数据库
     *
     * @param title 日期标题
     * @param targetDate 目标日期时间戳（毫秒）
     * @param category 类别名称（BIRTHDAY / ANNIVERSARY / HOLIDAY / OTHER）
     */
    fun createFirstDate(title: String, targetDate: Long, category: String) {
        if (title.isBlank()) return

        viewModelScope.launch {
            try {
                val now = System.currentTimeMillis()
                val date = SpecialDate(
                    title = title,
                    targetDate = targetDate,
                    category = category,
                    countMode = 0,
                    repeatType = 0,
                    reminderDays = 0,
                    isPinned = false,
                    isArchived = false,
                    cardStyle = "ORANGE_TEAR_OFF",
                    cardColor = "DEFAULT",
                    createdAt = now,
                    updatedAt = now
                )
                specialDateRepository.insert(date)
                _createdDateCount.value = _createdDateCount.value + 1
            } catch (e: Exception) {
                // 创建失败静默处理，用户可跳过
            }
        }
    }
```

```kotlin
/**
 * 获取今天 0 点的时间戳（毫秒）
 */
private fun todayStartMillis(): Long {
    val calendar = Calendar.getInstance()
    calendar.set(Calendar.HOUR_OF_DAY, 0)
    calendar.set(Calendar.MINUTE, 0)
    calendar.set(Calendar.SECOND, 0)
    calendar.set(Calendar.MILLISECOND, 0)
    return calendar.timeInMillis
}

/**
 * 格式化日期为 "2026/08/15 周六" 格式
 */
private fun formatDate(millis: Long): String {
    val dateFormat = SimpleDateFormat("yyyy/MM/dd EEE", Locale.CHINA)
    return dateFormat.format(Date(millis))
}

/**
 * 计算从今天到目标日期的天数差
 */
private fun daysUntil(targetMillis: Long): Int {
    val today = todayStartMillis()
    return ((targetMillis - today) / DAY_IN_MILLIS).toInt()
}
```

## 注意事项

1. **DateCategory 枚举**：`DateCategory` 定义在 `SpecialDateViewModel.kt`（line 997），不是 `DateCategory.kt`。直接 import 即可。
2. **SpecialDate 字段**：所有有 default 值的字段都可不传，使用默认值。`title`、`targetDate`、`createdAt`、`updatedAt` 是必填项。
3. **category 字段值**：存枚举名（如 `"BIRTHDAY"`）而非显示名（如"生日"）。
4. **DatePickerDialog**：Material3 的 `DatePickerDialog` 需要 `@OptIn(ExperimentalMaterial3Api::class)`。
5. **OutlinedCard onClick**：Material3 的 `OutlinedCard` 自带 `onClick` 参数（带涟漪效果），无需额外 clickable 修饰符。
6. **不要重复声明**：`SpecialDate`、`DateCategory`、`OnboardingViewModel` 等不要在 DateFeaturePage.kt 重复定义，只 import 使用。
