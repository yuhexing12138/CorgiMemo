### 步骤 3：实现最紧急任务判断逻辑

**判断优先级（从高到低）**：
1. `dueDate` 今天且 `priority=3`（高优先级）
2. `dueDate` 今天
3. `priority=3`
4. `priority=2`
5. 创建时间最早的待办

```kotlin
fun getMostUrgentTodo(todos: List<TodoItem>, currentTime: Long): TodoItem? {
    val pendingTodos = todos.filter { it.status == 0 }
    if (pendingTodos.isEmpty()) return null
    
    val todayStart = getTodayStart(currentTime)
    val todayEnd = getTodayEnd(currentTime)
    
    return pendingTodos
        .sortedWith(
            compareByDescending<TodoItem> { 
                // 截止日期今天且高优先级
                if (it.dueDate in todayStart..todayEnd && it.priority == 3) 4
                // 截止日期今天
                else if (it.dueDate in todayStart..todayEnd) 3
                // 高优先级
                else if (it.priority == 3) 2
                // 中优先级
                else if (it.priority == 2) 1
                else 0
            }.thenBy { it.createdAt }
        )
        .firstOrNull()
}
```

### 步骤 5：实现缓存机制

```kotlin
// 缓存数据结构
data class CachedGreeting(
    val greeting: String,
    val timeSlot: Int,        // 时间段索引（0-6）
    val timestamp: Long       // 缓存时间
)

// 30分钟 = 30 * 60 * 1000 = 1,800,000 毫秒
private const val CACHE_DURATION_MS = 30 * 60 * 1000L

// 判断是否需要刷新
fun shouldRefreshGreeting(cached: CachedGreeting?, currentTime: Long): Boolean {
    if (cached == null) return true
    
    // 检查时间段是否变化
    val currentTimeSlot = getTimeSlot(Calendar.getInstance().get(Calendar.HOUR_OF_DAY))
    if (currentTimeSlot != cached.timeSlot) return true
    
    // 检查缓存是否过期（30分钟）
    if (currentTime - cached.timestamp > CACHE_DURATION_MS) return true
    
    return false
}
```

### 步骤 6：创建天气管理器（WeatherManager）

由于 Android 没有标准系统天气 API，设计为**可插拔接口**：

```kotlin
interface WeatherProvider {
    suspend fun getCurrentWeather(): WeatherInfo?
}

// 占位实现：始终返回 UNKNOWN
class DefaultWeatherProvider : WeatherProvider {
    override suspend fun getCurrentWeather(): WeatherInfo? = null
}

// WeatherManager 包装层
object WeatherManager {
    private var provider: WeatherProvider = DefaultWeatherProvider()
    
    fun setProvider(provider: WeatherProvider) {
        this.provider = provider
    }
    
    suspend fun getWeatherInfo(): WeatherInfo? {
        return try {
            provider.getCurrentWeather()
        } catch (e: Exception) {
            null  // 获取失败时降级
        }
    }
    
    // 天气提示语
    fun getWeatherMessage(weather: WeatherInfo): String = when (weather.condition) {
        WeatherCondition.SUNNY -> "☀️ 今天天气不错"
        WeatherCondition.RAINY -> "🌧️ 记得带伞哦"
        WeatherCondition.CLOUDY -> "☁️ 多云天气"
        WeatherCondition.SNOWY -> "❄️ 下雪啦，注意保暖"
        WeatherCondition.UNKNOWN -> ""
    }
}
```

### 步骤 7：集成到 HomeViewModel

修改 `HomeViewModel.kt`：

```kotlin
class HomeViewModel @Inject constructor(...) : ViewModel() {
    
    // 新增：问候语缓存
    private var cachedGreeting: CachedGreeting? = null
    
    // 新增：动态问候语方法（每次进入首页调用）
    suspend fun refreshGreetingIfNeeded() {
        val currentTime = System.currentTimeMillis()
        
        if (shouldRefreshGreeting(cachedGreeting, currentTime)) {
            val greeting = generateDynamicGreeting(currentTime)
            _greeting.value = greeting
            cachedGreeting = CachedGreeting(
                greeting = greeting,
                timeSlot = getCurrentTimeSlot(),
                timestamp = currentTime
            )
        }
    }
    
    // 生成动态问候语
    private suspend fun generateDynamicGreeting(currentTime: Long): String {
        // 1. 获取基础数据
        val calendar = Calendar.getInstance()
        calendar.timeInMillis = currentTime
        val hour = calendar.get(Calendar.HOUR_OF_DAY)
        val dayOfWeek = calendar.get(Calendar.DAY_OF_WEEK)
        val isWeekend = dayOfWeek == Calendar.SATURDAY || dayOfWeek == Calendar.SUNDAY
        
        // 2. 获取待办数据
        val todos = todoRepository.getAllTodos().first()
        val pendingCount = todos.count { it.status == 0 }
        val urgentTodo = getMostUrgentTodo(todos, currentTime)
        
        // 3. 获取天气信息（失败不影响）
        val weatherInfo = WeatherManager.getWeatherInfo()
        
        // 4. 获取用户名（使用柯基名字或新增字段）
        val userName = corgiData.value?.name
        
        // 5. 构建上下文并生成问候语
        val context = GreetingContext(
            hour = hour,
            isWeekend = isWeekend,
            pendingTodoCount = pendingCount,
            urgentTodoTitle = urgentTodo?.title,
            userName = userName,
            weatherInfo = weatherInfo
        )
        
        return DynamicGreetingManager.generateGreeting(context)
    }
}
```

## 四、潜在依赖和考量

### 4.1 用户名决策

| 方案 | 优点 | 缺点 | 建议 |
|------|------|------|------|
| 使用柯基名字 | 无需修改现有代码 | 模板中"用户名"变为"柯基名" | ⭐ 推荐（最小改动） |
| 新增用户名字段 | 语义正确 | 需要修改 DataStore、添加设置入口 | 如需真实用户名可后续迭代 |
| 移除用户名 | 最简单 | 失去个性化 | 不推荐 |

**建议**：先使用柯基名字作为占位，模板文案调整为更自然的表达，如：
```
"上午好！今天有 5 个待办，最紧急的是「提交报告」，加油💪"
```

### 4.2 天气功能决策

| 方案 | 实现复杂度 | 依赖 | 建议 |
|------|------------|------|------|
| 第三方 API（如和风天气） | 中 | 需要 API Key、网络权限 | 后续版本实现 |
| 系统 Weather Provider | 高 | 仅部分厂商支持 | 不推荐 |
| 预留接口 + Mock | 低 | 无 | ⭐ 推荐（当前版本） |

**建议**：当前版本实现 `WeatherManager` 接口框架，使用占位实现返回 `null`，天气信息不显示。待后续确定天气服务后再接入。
