```kotlin
        Calendar.SATURDAY -> 7
        Calendar.SUNDAY -> 6

    cal.set(Calendar.MINUTE, 0)
```

### 步骤 9：通知聚合逻辑（在 NotificationHelper 中实现）

```kotlin
// 使用 SharedPreferences 追踪同一小时内的通知数量
private const val PREF_NOTIFICATION_TRACKER = "notification_tracker"
private const val KEY_HOUR_START = "hour_start"
private const val KEY_COUNT = "count"
private const val KEY_TODO_IDS = "todo_ids"

/**
 * 检查是否需要聚合通知
 *
 * @return Pair<是否需要聚合, 聚合的待办ID列表>
 */
private fun shouldGroupNotifications(
    context: Context,
    todoId: Long
): Pair<Boolean, List<Long>> {
    val prefs = context.getSharedPreferences(
        PREF_NOTIFICATION_TRACKER,
        Context.MODE_PRIVATE
    )
  
    val currentHourStart = getCurrentHourStart()
    val savedHourStart = prefs.getLong(KEY_HOUR_START, 0)
    val count = prefs.getInt(KEY_COUNT, 0)
    val todoIds = prefs.getStringSet(KEY_TODO_IDS, emptySet()) ?: emptySet()
  
    // 如果是新的一小时，重置计数器
    if (currentHourStart != savedHourStart) {
        prefs.edit()
            .putLong(KEY_HOUR_START, currentHourStart)
            .putInt(KEY_COUNT, 1)
            .putStringSet(KEY_TODO_IDS, setOf(todoId.toString()))
            .apply()
        return Pair(false, listOf(todoId))
    }
  
    // 同一小时内，累加计数
    val newCount = count + 1
    val newTodoIds = todoIds + todoId.toString()
  
    prefs.edit()
        .putInt(KEY_COUNT, newCount)
        .putStringSet(KEY_TODO_IDS, newTodoIds)
        .apply()
  
    // 如果达到 3 条，需要聚合
    return if (newCount >= 3) {
        Pair(true, newTodoIds.map { it.toLong() })
    } else {
        Pair(false, listOf(todoId))
    }
}

/**
 * 获取当前小时开始的时间戳
 */
private fun getCurrentHourStart(): Long {
    val cal = Calendar.getInstance()
    cal.set(Calendar.MINUTE, 0)
    cal.set(Calendar.SECOND, 0)
    cal.set(Calendar.MILLISECOND, 0)
    return cal.timeInMillis
}
```

## 四、潜在依赖和考量

### 4.1 通知权限

| Android 版本          | 权限要求                               |
| --------------------- | -------------------------------------- |
| Android 13+ (API 33)  | 需要 `POST_NOTIFICATIONS` 运行时权限 |
| Android 8.0+ (API 26) | 需要创建通知渠道                       |
| Android 8.0 以下      | 无需特殊处理                           |

**当前状态**：GeofenceRepository 中已有权限检查逻辑，可以复用。

### 4.2 AlarmManager 精准性

| Android 版本          | 闹钟限制                                                               |
| --------------------- | ---------------------------------------------------------------------- |
| Android 12+ (API 31)  | 需要 `SCHEDULE_EXACT_ALARM` 权限或使用 `setExactAndAllowWhileIdle` |
| Android 6.0+ (API 23) | 低电耗模式影响普通闹钟，需使用 `setAndAllowWhileIdle`                |
| Android 5.1 以下      | 无特殊限制                                                             |

**建议**：使用 `setExactAndAllowWhileIdle` 确保提醒准时。

### 4.3 自定义分类的提醒提前量

| 方案           | 说明                                 | 建议                   |
| -------------- | ------------------------------------ | ---------------------- |
| 统一默认值     | 所有自定义分类默认 30 分钟           | ⭐ 推荐                |
| 按分类单独设置 | 每个自定义分类可单独设置             | 后续优化               |
| 关联到类型     | 自定义分类可选择"工作/学习/生活"类型 | 需要修改 Category 模型 |

**当前方案**：自定义分类默认 30 分钟，支持单独修改。

### 4.4 通知聚合的显示效果

**Android 7.0+** 支持通知分组（Notification Group）：

- 子通知：单条待办详情
- 总结通知："你有 N 个待办即将到期"
- 用户展开分组后可看到所有待办

**Android 6.0 及以下**：

- 不支持分组，降级为单条通知
- 聚合时可使用 InboxStyle 显示多条内容

| 同一待办设置多个闹钟 | 使用 `todoId` 作为 `requestCode`，确保不会重复 |
| 通知聚合时待办已完成 | 从聚合列表中排除已完成的待办                       |
| 跨小时的通知聚合     | 每小时重置计数器，不跨小时聚合                     |
