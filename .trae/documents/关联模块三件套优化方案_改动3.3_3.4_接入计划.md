**文件路径**：`c:\Users\EDY\Desktop\CorgiMemo\app\src\main\java\com\corgimemo\app\ui\components\RelationListBottomSheet.kt`

**设计要点**：

| 维度 | 设计 |
| --- | --- |
| 形态 | ModalBottomSheet（与 RelationPickerBottomSheet 风格一致） |
| 数据源 | `RelationListViewModel.relations + titles + isLoading` |
| 列表项 | 复用 `LinkedCardPreviewDialog` 中提取的视觉风格：类型图标 + 标题 + 类型标签 |
| 交互 | 点击列表项 → 触发 `onItemClick(cardType, cardId)` 让父级跳转详情页 |
| 解绑 | 每项右侧显示 × 按钮，调用 `viewModel.unlink(relationId)` |
| 关闭 | 弹窗 onDismissRequest 调 `viewModel.clear()` |
| 关联后刷新 | 弹窗关闭后调 `onDismiss()` 回调，外层 ViewModel 重新查 `relationCountMap` |

**关键代码骨架**（重点列出，详细实现写到文件）：

```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun RelationListBottomSheet(
    visible: Boolean,
    sourceType: String,       // "todo" | "inspiration" | "date"
    sourceId: Long,
    groupId: Int = 0,
    onItemClick: (cardType: String, cardId: Long) -> Unit = {},  // 跳转详情
    onDismiss: () -> Unit = {},                                 // 用于通知外层刷新
    viewModel: RelationListViewModel = hiltViewModel()
) {
    val sheetState = rememberModalBottomSheetState(skipPartiallyExpanded = true)
    val relations by viewModel.relations.collectAsState()
    val titles by viewModel.titles.collectAsState()
    val isLoading by viewModel.isLoading.collectAsState()

    // 弹窗打开时加载数据
    LaunchedEffect(visible, sourceType, sourceId, groupId) {
        if (visible) {
            viewModel.loadRelations(sourceType, sourceId, groupId)
        } else {
            viewModel.clear()
        }
    }

    if (!visible) return

    ModalBottomSheet(
        onDismissRequest = onDismiss,
        sheetState = sheetState,
        containerColor = MaterialTheme.colorScheme.surface
    ) {
        Column(modifier = Modifier.fillMaxWidth().padding(horizontal = 16.dp, vertical = 8.dp)) {
            // 标题
            Text(text = "关联卡片（${relations.size}）", fontSize = 16.sp, fontWeight = FontWeight.SemiBold)
            Spacer(Modifier.height(12.dp))

            when {
                isLoading -> {
                    Box(Modifier.fillMaxWidth().padding(32.dp), contentAlignment = Alignment.Center) {
                        CircularProgressIndicator()
                    }
                }
                relations.isEmpty() -> {
                    Text("暂无关联卡片", fontSize = 14.sp, color = Color(0xFF999999), modifier = Modifier.padding(16.dp))
                }
                else -> {
                    relations.forEach { relation ->
                        RelationListItem(
                            relation = relation,
                            targetTitle = titles[relation.id] ?: "已删除",
                            onClick = { onItemClick(relation.targetType, relation.targetId) },
                            onUnlink = { viewModel.unlink(relation.id) }
                        )
                    }
                }
            }

            Spacer(Modifier.height(16.dp))
        }
    }
}

@Composable
private fun RelationListItem(
    relation: CardRelation,
    targetTitle: String,
    onClick: () -> Unit,
    onUnlink: () -> Unit
) {
    val (typeBg, typeIconText, typeLabel) = when (relation.targetType) {
        "todo" -> Triple(Color(0xFFE3F2FD), "📝", "待办")
        "inspiration" -> Triple(Color(0xFFFFF3E0), "💡", "灵感")
        "date" -> Triple(Color(0xFFFCE4EC), "📅", "日期")
        else -> Triple(Color(0xFFF5F5F5), "📎", "未知")
    }

    Row(
        modifier = Modifier
            .fillMaxWidth()
            .clickable(onClick = onClick)
            .padding(vertical = 10.dp),
        verticalAlignment = Alignment.CenterVertically
    ) {
        Box(
            modifier = Modifier
                .size(36.dp)
                .clip(RoundedCornerShape(8.dp))
                .background(typeBg),
            contentAlignment = Alignment.Center
        ) {
            Text(typeIconText, fontSize = 16.sp)
        }
        Spacer(Modifier.width(12.dp))
        Column(Modifier.weight(1f)) {
            Text(targetTitle, fontSize = 14.sp, color = Color(0xFF2D2D2D), maxLines = 1, overflow = TextOverflow.Ellipsis)
            Text(typeLabel, fontSize = 11.sp, color = Color(0xFF999999))
        }
        IconButton(onClick = onUnlink) {
            Icon(Icons.Outlined.Close, contentDescription = "解除关联", tint = Color(0xFF999999))
        }
    }
}
```

**颜色规范**：参考 `LinkedCardPreviewDialog.kt` 的类型色配置（E3F2FD/FFF3E0/FCE4EC）。

### 改动 3.5：解除关联后刷新外层

**为什么需要**：
解除关联后，DB 写入成功，但外层 `relationCountMap` 是 StateFlow 缓存，**不会自动更新**。关闭弹窗时需触发外层重新查询。

**方案**：
- `RelationListBottomSheet` 接受 `onRefreshNeeded: () -> Unit = {}` 回调
- 解除关联成功 → 调 `onRefreshNeeded`
- 外层（HomeScreen/InspirationScreen）的 onRefreshNeeded 调对应 ViewModel 的 `refreshRelationCounts(currentList)`

**HomeViewModel 需新增公共方法**：
```kotlin
fun refreshRelationCountsOnDemand() {
    val allTodos = (pinnedPendingTodos.value + pendingTodos.value + 
                    pinnedCompletedTodos.value + completedTodos.value)
    viewModelScope.launch { refreshRelationCounts(allTodos) }
}
```

**InspirationViewModel 同理**：抽 `refreshRelationCountsOnDemand()`，从 `_allInspirations` 取数据。
