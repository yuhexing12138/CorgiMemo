```kotlin
@Composable
fun CornerDecoration(
    modifier: Modifier = Modifier,
    color: Color = Color(0xFFF97316), // 暖橙色
    size: Dp = 24.dp,
    strokeWidth: Dp = 3.dp
) {
    Canvas(modifier = modifier.size(size)) {
        // 绘制 L 型对角条纹
        drawLine(
            color = color,
            start = Offset(0f, strokeWidth.toPx()),
            end = Offset(size.toPx(), strokeWidth.toPx()),
            strokeWidth = strokeWidth.toPx()
        )
        drawLine(
            color = color,
            start = Offset(strokeWidth.toPx(), 0f),
            end = Offset(strokeWidth.toPx(), size.toPx()),
            strokeWidth = strokeWidth.toPx()
        )
    }
}

// 使用方式
Box(modifier = Modifier.fillMaxSize()) {
    // 内容区域
    Column(modifier = Modifier.padding(48.dp)) { /* ... */ }

    // 四角装饰

        modifier = Modifier.align(Alignment.TopStart).padding(8.dp)
    )
    CornerDecoration(
        modifier = Modifier.align(Alignment.TopEnd).padding(8.dp)
    )
    // ... 其他两角
}

@Composable
fun EditToolbar(
    onPhotoClick: () -> Unit,
    onTextFormatClick: () -> Unit,
    onBoldClick: () -> Unit,
    onListClick: () -> Unit,
    onTodoClick: () -> Unit,
    onBackgroundClick: () -> Unit
) {
    Surface(
        shape = RoundedCornerShape(topStart = 16.dp, topEnd = 16.dp),
        shadowElevation = 8.dp
    ) {
        Row(
            modifier = Modifier
                .fillMaxWidth()
                .padding(horizontal = 16.dp, vertical = 12.dp),
            horizontalArrangement = Arrangement.SpaceEvenly
        ) {
            ToolButton(icon = "📷", label = "照片", onClick = onPhotoClick)
            ToolButton(icon = "📝", label = "文本", onClick = onTextFormatClick)
            ToolButton(icon = "**B**", label = "加粗", onClick = onBoldClick)
            ToolButton(icon = "☰", label = "列表", onClick = onListClick)
            ToolButton(icon = "☑️", label = "待办", onClick = onTodoClick)
            ToolButton(icon = "🎨", label = "背景", onClick = onBackgroundClick)
        }
    }
}

@Composable
private fun ToolButton(
    icon: String,
    label: String,
    onClick: () -> Unit
) {
    Column(
        horizontalAlignment = Alignment.CenterHorizontally,
        modifier = Modifier
            .clickable(onClick = onClick)
            .padding(8.dp)
    ) {
        Text(text = icon, fontSize = 22.sp)
        Spacer(modifier = Modifier.height(4.dp))
        Text(text = label, fontSize = 11.sp, color = Color.Gray)
    }
}
```
