### 步骤 2：创建节气管理器

**文件**: `SolarTermManager.kt`

```kotlin
/**
 * 节气管理器
 * 参考 HolidayManager 设计模式
 */
object SolarTermManager {

    /**
     * 获取当前节气
     *
     * @param currentTime 当前时间戳
     * @return 当前节气对象，如果不是节气当天返回 null
     */
    fun getCurrentSolarTerm(currentTime: Long = System.currentTimeMillis()): SolarTerm? {
        val calendar = Calendar.getInstance(TimeZone.getDefault())
        calendar.timeInMillis = currentTime
        val month = calendar.get(Calendar.MONTH) + 1
        val day = calendar.get(Calendar.DAY_OF_MONTH)

        return SolarTermData.allSolarTerms.find { solarTerm ->
            isSolarTerm(solarTerm, month, day)
        }
    }

    /**
     * 判断某天是否为某个节气
     *
     * @param solarTerm 节气
     * @param month 月份（1-12）
     * @param day 日期
     * @return 是否为该节气
     */
    fun isSolarTerm(solarTerm: SolarTerm, month: Int, day: Int): Boolean {
        return solarTerm.date.month == month && day in solarTerm.date.dayRange
    }

    /**
     * 根据 ID 获取节气
     */
    fun getSolarTermById(id: String): SolarTerm? {
        return SolarTermData.allSolarTerms.find { it.id == id }
    }

    /**
     * 获取下一个即将到来的节气
     *
     * @param currentTime 当前时间戳
     * @return 下一个节气和距离天数的 Pair
     */
    fun getNextSolarTerm(currentTime: Long = System.currentTimeMillis()): Pair<SolarTerm, Int>? {
        val calendar = Calendar.getInstance(TimeZone.getDefault())
        calendar.timeInMillis = currentTime
        val currentYear = calendar.get(Calendar.YEAR)
        val currentMonth = calendar.get(Calendar.MONTH) + 1
        val currentDay = calendar.get(Calendar.DAY_OF_MONTH)

        // 检查今年的所有节气
        var nextSolarTerm: SolarTerm? = null
        var minDaysDiff = Int.MAX_VALUE

        for (solarTerm in SolarTermData.allSolarTerms) {
            // 简化处理：假设节气日期固定在每月的中间值
            // 实际项目中可能需要更精确的日期计算
            val solarTermMonth = solarTerm.date.month
            val solarTermDay = solarTerm.date.dayRange.first + 
                (solarTerm.date.dayRange.last - solarTerm.date.dayRange.first) / 2

            var daysDiff = 0
            if (solarTermMonth > currentMonth ||
                (solarTermMonth == currentMonth && solarTermDay >= currentDay)) {
                // 今年的节气还没到
                val monthDiff = solarTermMonth - currentMonth
                daysDiff = monthDiff * 30 + (solarTermDay - currentDay)
                if (daysDiff < minDaysDiff && daysDiff >= 0) {
                    minDaysDiff = daysDiff
                    nextSolarTerm = solarTerm
                }
            }
        }

        // 如果今年的节气都过了，取明年第一个
        if (nextSolarTerm == null) {
            val firstSolarTerm = SolarTermData.allSolarTerms.first()
            nextSolarTerm = firstSolarTerm
            minDaysDiff = 365 // 简化处理
        }

        return nextSolarTerm?.let { Pair(it, minDaysDiff) }
    }
}
```

```kotlin
// 添加到 companion object 中
companion object {
    // 已有键...
    
    // 节气卡片关闭状态
    // key 格式: "solar_term_dismissed_YYYYMMDD"
    private fun getSolarTermDismissedKey(dateKey: String) =
        booleanPreferencesKey("solar_term_dismissed_$dateKey")
}

/**
 * 获取今天的日期 Key（YYYYMMDD 格式）
 */
private fun getTodayKey(): String {
    val cal = Calendar.getInstance()
    val year = cal.get(Calendar.YEAR)
    val month = cal.get(Calendar.MONTH) + 1
    val day = cal.get(Calendar.DAY_OF_MONTH)
    return String.format("%04d%02d%02d", year, month, day)
}

/**
 * 检查今天节气卡片是否已关闭
 *
 * @return 是否已关闭
 */
suspend fun isSolarTermCardDismissed(): Boolean {
    val key = getSolarTermDismissedKey(getTodayKey())
    return dataStore.data.map { prefs ->
        prefs[key] ?: false
    }.first()
}

/**
 * 标记今天的节气卡片为已关闭
 */
suspend fun dismissSolarTermCard() {
    val key = getSolarTermDismissedKey(getTodayKey())
    dataStore.edit { prefs ->
        prefs[key] = true
    }
}
```

## 五、潜在依赖和考量

### 5.1 分享功能方案对比

| 方案 | 优点 | 缺点 | 建议 |
|------|------|------|------|
| **文本分享** | 简单可靠，无需额外权限 | 视觉效果一般 | ⭐ 推荐（首选） |
| **图片分享** | 视觉效果好，支持朋友圈 | 需要截图权限，实现复杂 | 后续扩展 |
| **混合方案** | 先文本分享，再升级图片 | - | 渐进式实现 |

**当前方案**：先实现文本分享，后续可扩展图片分享。

### 5.2 节气日期精度

**问题**：24节气的日期每年略有不同（如立春可能是2月3、4、5日）。

**解决方案**：
- 当前方案：使用 `IntRange` 范围判断，可覆盖绝大多数情况
- 精确方案：引入节气算法（如《定气法》计算太阳黄经）

**建议**：使用范围判断已足够，后续如有精确需求可升级。

### 5.3 同一天既是节日又是节气

| 优先级 | 场景 | 显示内容 |
|--------|------|----------|
| 1 | 节气当天 | 节气科普卡片 + 柯基卡片（节日问候语嵌入柯基卡片） |
| 2 | 节日当天（非节气） | 柯基卡片（节日问候语嵌入） |

**理由**：节气科普卡片有独立的视觉区域，不影响节日装扮和问候语。

### 5.4 滑动关闭的持久化

| 维度 | 说明 |
|------|------|
| Key 格式 | `solar_term_dismissed_YYYYMMDD` |
| 有效期 | 当天有效，第二天自动重置 |
| 存储位置 | DataStore（本地持久化） |

## 六、风险和边界情况处理

| 风险场景 | 处理方案 |
|----------|----------|

| 节气卡片分享内容超长 | 限制科普知识在 100 字以内 |
| 滑动关闭误操作 | 需要滑动足够距离才触发，且可撤销（可选） |
