@Dao
interface AchievementDao {
    // 查询所有
    @Query("SELECT * FROM achievements")
    suspend fun getAll(): List<AchievementEntity>

    // 查询已解锁
    @Query("SELECT * FROM achievements WHERE unlockedAt IS NOT NULL ORDER BY unlockedAt DESC")
    suspend fun getUnlocked(): List<AchievementEntity>

    // 查询单个
    @Query("SELECT * FROM achievements WHERE id = :id")
    suspend fun getById(id: String): AchievementEntity?

    // 插入（忽略已存在）
    @Insert(onConflict = OnConflictStrategy.IGNORE)
    suspend fun insert(achievement: AchievementEntity)

    @Insert(onConflict = OnConflictStrategy.IGNORE)
    suspend fun insertAll(achievements: List<AchievementEntity>)

    // 解锁（更新时间戳）
    @Query("UPDATE achievements SET unlockedAt = :timestamp WHERE id = :id")
    suspend fun unlock(id: String, timestamp: Long)

    // 检查是否已解锁
    @Query("SELECT COUNT(*) > 0 FROM achievements WHERE id = :id AND unlockedAt IS NOT NULL")
    suspend fun isUnlocked(id: String): Boolean

    // 获取已解锁数量
    @Query("SELECT COUNT(*) FROM achievements WHERE unlockedAt IS NOT NULL")
    suspend fun getUnlockedCount(): Int

    // 获取已解锁 ID 列表
    @Query("SELECT id FROM achievements WHERE unlockedAt IS NOT NULL")
    suspend fun getUnlockedIds(): List<String>
}
```

**注意事项**:
- 所有查询都是 suspend 函数
- 插入使用 `OnConflictStrategy.IGNORE` 避免重复插入

            modifier = Modifier.fillMaxWidth(),

                isActive = currentStage == AchievementStage.BEGINNER,
                isPast = false
            )
            // ... 其他阶段

        // 进度条
        val progress = unlockedCount.toFloat() / totalCount.toFloat()
        LinearProgressIndicator(
            progress = progress,
            modifier = Modifier.fillMaxWidth().height(8.dp).clip(RoundedCornerShape(4.dp)),
            color = Color(currentStage.color)

        Text(

            fontSize = 12.sp,

        )

private suspend fun unlockAchievement(achievementId: String) {
    // 1. 解锁成就
    val achievement = achievementRepository.unlock(achievementId) ?: return

    // 2. 检查是否有对应的装扮需要解锁
    val outfitId = AchievementManager.getOutfitForAchievement(achievementId)
    if (outfitId != null) {
        val corgiData = corgiRepository.getCorgiData()
        if (corgiData != null && !OutfitManager.isOutfitUnlocked(outfitId, corgiData.unlockedOutfits)) {
            val updatedOutfits = OutfitManager.addOutfits(
                corgiData.unlockedOutfits,
                listOf(outfitId)
            )
            corgiRepository.updateUnlockedOutfits(updatedOutfits)
        }
    }

    // 3. 发送事件
    _achievementUnlockEvents.emit(achievement)

    // 4. 检测复合成就
    checkAllAchievements()
}
```
